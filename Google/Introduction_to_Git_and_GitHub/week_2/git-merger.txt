git commit -a -m "Call check_reboot from main, exit with 1 on error" - <-a> ярлык для выполнения уже добавленого файла
git log -p - флаг <-p>  позволяет уидеть полее подробную информацию всех фиксаций
git show <id commit> - подробная информация о фиксации по номеру
git log --stat - какие файлы были именены и сколько строк было добавлено и удалено
git add -p - показывает изменения и задает вопрос хоти ли мы их записать
git diff --staged - показывает изменения которые записаны но не зафиксированы
git rm - удаляет файл из репозитория
git mv - переименовывает файл
.gitignore - игнорировать ненужные файлы
git checkout|(restore) <file> - отменяет изменения внесенные в файл
git reset HEAD <file...> - отменяем добавления из промежуточной области
git commit --amend - перезаписывает предыдущую фиксацию
git revert HEAD - откатить фиксацию на прошлую версию
git revert <id> - откат выбранной фиксации по id
